import React, { useState, useEffect, useRef } from 'react';

import { 

  Play, 

  Pause, 

  RotateCcw, 

  Plus, 

  Trash2, 

  ChevronUp, 

  ChevronDown, 

  Volume2, 

  VolumeX,

  Clock,

  Copy,

  Zap,

  Save,

  Download

} from 'lucide-react';


const DEFAULT_SEQUENCE = [

  { id: '1', label: 'GET READY', duration: 15, remaining: 15 },

  { id: '2', label: 'SET 1', duration: 60, remaining: 60 },

  { id: '3', label: 'REST', duration: 30, remaining: 30 },

  { id: '4', label: 'SET 2', duration: 60, remaining: 60 },

  { id: '5', label: 'REST', duration: 30, remaining: 30 },

  { id: '6', label: 'SET 3', duration: 60, remaining: 60 },

  { id: '7', label: 'REST', duration: 30, remaining: 30 },

];


const App = () => {

  const [timers, setTimers] = useState(DEFAULT_SEQUENCE);

  const [savedSequences, setSavedSequences] = useState([]);

  const [saveName, setSaveName] = useState('');

  

  const [currentIndex, setCurrentIndex] = useState(null);

  const [isActive, setIsActive] = useState(false);

  const [isMuted, setIsMuted] = useState(false);

  

  const intervalRef = useRef(null);

  const audioContextRef = useRef(null);


  const initAudio = () => {

    if (!audioContextRef.current) {

      audioContextRef.current = new (window.AudioContext || window.webkitAudioContext)();

    }

  };


  // More aggressive "Buzzer" sound

  const playAlert = async (type = 'transition') => {

    if (isMuted || !audioContextRef.current) return;

    

    const playBuzzer = (freq1, freq2, duration, delay) => {

      setTimeout(() => {

        const osc1 = audioContextRef.current.createOscillator();

        const osc2 = audioContextRef.current.createOscillator();

        const gain = audioContextRef.current.createGain();

        

        // Mixing a sawtooth and square wave at low, slightly detuned frequencies

        // creates a harsh, metallic "sports scoreboard" buzzer sound.

        osc1.type = 'sawtooth'; 

        osc2.type = 'square'; 

        osc1.frequency.setValueAtTime(freq1, audioContextRef.current.currentTime);

        osc2.frequency.setValueAtTime(freq2, audioContextRef.current.currentTime);

        

        gain.gain.setValueAtTime(0.3, audioContextRef.current.currentTime);

        // Linear ramp gives it a more mechanical, abrupt cut-off

        gain.gain.linearRampToValueAtTime(0.01, audioContextRef.current.currentTime + duration);

        

        osc1.connect(gain);

        osc2.connect(gain);

        gain.connect(audioContextRef.current.destination);

        

        osc1.start();

        osc2.start();

        osc1.stop(audioContextRef.current.currentTime + duration);

        osc2.stop(audioContextRef.current.currentTime + duration);

      }, delay);

    };


    if (type === 'transition') {

      // Sharp, short double-buzz for transitions

      playBuzzer(120, 125, 0.2, 0);

      playBuzzer(120, 125, 0.3, 300);

    } else {

      // Long, deep, sustained buzzer for the final finish

      playBuzzer(100, 105, 1.5, 0);

    }

  };


  useEffect(() => {

    if (isActive && currentIndex !== null) {

      intervalRef.current = setInterval(() => {

        setTimers(prev => {

          const newTimers = [...prev];

          const current = newTimers[currentIndex];

          

          if (current.remaining > 0) {

            newTimers[currentIndex] = { ...current, remaining: current.remaining - 1 };

            return newTimers;

          } else {

            clearInterval(intervalRef.current);

            

            if (currentIndex < timers.length - 1) {

              playAlert('transition');

              setCurrentIndex(currentIndex + 1);

            } else {

              playAlert('finish');

              setIsActive(false);

              setCurrentIndex(null);

              return prev.map(t => ({ ...t, remaining: t.duration }));

            }

            return newTimers;

          }

        });

      }, 1000);

    } else {

      clearInterval(intervalRef.current);

    }

    return () => clearInterval(intervalRef.current);

  }, [isActive, currentIndex, timers.length, isMuted]);


  const handleStart = () => {

    initAudio();

    if (currentIndex === null) setCurrentIndex(0);

    setIsActive(true);

  };


  const handlePause = () => setIsActive(false);


  const handleReset = () => {

    setIsActive(false);

    setCurrentIndex(null);

    setTimers(timers.map(t => ({ ...t, remaining: t.duration })));

  };


  const saveSequence = () => {

    if (!saveName.trim() || timers.length === 0) return;

    const newSaved = {

      id: Date.now().toString(),

      name: saveName.toUpperCase(),

      timers: timers.map(t => ({...t, remaining: t.duration}))

    };

    setSavedSequences([...savedSequences, newSaved]);

    setSaveName('');

  };


  const loadSequence = (sequenceToLoad) => {

    if (isActive) return;

    const newTimers = sequenceToLoad.map(t => ({

       ...t, 

       id: `${Date.now()}-${Math.random().toString(36).substr(2, 9)}`,

       remaining: t.duration 

    }));

    setTimers(newTimers);

    setCurrentIndex(null);

    setIsActive(false);

  };


  const addTimer = () => {

    const newTimer = {

      id: Date.now().toString(),

      label: 'NEW DRILL',

      duration: 60,

      remaining: 60

    };

    setTimers([...timers, newTimer]);

  };


  const duplicateTimer = (id) => {

    if (isActive) return;

    const index = timers.findIndex(t => t.id === id);

    if (index === -1) return;

    const timerToCopy = timers[index];

    const newTimer = {

      ...timerToCopy,

      id: `${Date.now()}-${Math.random().toString(36).substr(2, 9)}`,

      remaining: timerToCopy.duration

    };

    const newTimers = [...timers];

    newTimers.splice(index + 1, 0, newTimer);

    setTimers(newTimers);

  };


  const removeTimer = (id) => {

    if (isActive) return;

    setTimers(timers.filter(t => t.id !== id));

  };


  const updateTimer = (id, field, value) => {

    setTimers(timers.map(t => {

      if (t.id === id) {

        const val = field === 'duration' ? Math.max(1, parseInt(value) || 0) : value;

        return { ...t, [field]: val, remaining: field === 'duration' ? val : t.remaining };

      }

      return t;

    }));

  };


  const moveTimer = (index, direction) => {

    if (isActive) return;

    const newTimers = [...timers];

    const targetIndex = index + direction;

    if (targetIndex < 0 || targetIndex >= timers.length) return;

    [newTimers[index], newTimers[targetIndex]] = [newTimers[targetIndex], newTimers[index]];

    setTimers(newTimers);

  };


  const formatTime = (seconds) => {

    const mins = Math.floor(seconds / 60);

    const secs = seconds % 60;

    return `${mins}:${secs.toString().padStart(2, '0')}`;

  };


  const totalTimeRemaining = timers.reduce((acc, curr, idx) => {

    if (currentIndex !== null && idx < currentIndex) return acc;

    return acc + curr.remaining;

  }, 0);


  return (

    <div className="min-h-screen bg-black p-4 md:p-8 font-sans text-white uppercase tracking-tighter">

      <div className="max-w-2xl mx-auto">

        {/* Header */}

        <header className="flex justify-between items-center mb-10 border-b-4 border-amber-500 pb-4">

          <div>

            <h1 className="text-4xl font-black italic flex items-center gap-2 text-amber-500">

              <Zap fill="currentColor" className="text-amber-500" />

              IRON SEQUENCE

            </h1>

            <p className="text-neutral-500 text-xs font-bold tracking-[0.2em]">High Performance Training</p>

          </div>

          <button 

            onClick={() => setIsMuted(!isMuted)}

            className="p-2 rounded-full hover:bg-neutral-800 transition-colors text-amber-500"

          >

            {isMuted ? <VolumeX size={28} /> : <Volume2 size={28} />}

          </button>

        </header>


        {/* Main Display */}

        <div className="bg-neutral-900 rounded-lg shadow-[0_0_50px_rgba(245,158,11,0.15)] p-10 mb-10 text-center border-2 border-neutral-800 relative overflow-hidden">

          <div className="absolute top-0 left-0 w-full h-1 bg-neutral-800">

            {currentIndex !== null && (

              <div 

                className="h-full bg-amber-500 shadow-[0_0_15px_rgba(245,158,11,0.8)] transition-all duration-1000"

                style={{ width: `${((timers[currentIndex].duration - timers[currentIndex].remaining) / timers[currentIndex].duration) * 100}%` }}

              />

            )}

          </div>


          <div className="text-amber-500 font-black text-xl mb-4 tracking-widest">

            {currentIndex !== null ? timers[currentIndex].label : "READY TO WORK"}

          </div>

          <div className="text-8xl md:text-9xl font-black tabular-nums mb-8 leading-none">

            {currentIndex !== null ? formatTime(timers[currentIndex].remaining) : formatTime(timers[0]?.duration || 0)}

          </div>

          

          <div className="flex justify-center gap-6">

            {!isActive ? (

              <button 

                onClick={handleStart}

                className="flex items-center gap-3 bg-amber-500 hover:bg-amber-400 text-black px-12 py-5 rounded-md font-black text-2xl transition-all transform hover:scale-105 active:scale-95 shadow-lg shadow-amber-900/20"

              >

                <Play fill="currentColor" size={28} /> GO

              </button>

            ) : (

              <button 

                onClick={handlePause}

                className="flex items-center gap-3 bg-white hover:bg-neutral-200 text-black px-12 py-5 rounded-md font-black text-2xl transition-all transform hover:scale-105 active:scale-95 shadow-lg"

              >

                <Pause fill="currentColor" size={28} /> STOP

              </button>

            )}

            <button 

              onClick={handleReset}

              className="px-6 rounded-md border-2 border-neutral-700 text-neutral-500 hover:text-white hover:border-white transition-all active:scale-90"

            >

              <RotateCcw size={28} />

            </button>

          </div>

          

          <div className="mt-8 text-neutral-500 font-bold tracking-widest text-xs">

            TOTAL TIME LEFT: {formatTime(totalTimeRemaining)}

          </div>

        </div>


        {/* List of Timers */}

        <div className="space-y-4">

          <div className="flex justify-between items-end border-b border-neutral-800 pb-2">

            <h2 className="font-black text-neutral-400 text-lg tracking-widest">DRILL LIST</h2>

            <button 

              onClick={addTimer}

              disabled={isActive}

              className="flex items-center gap-1 text-sm font-black text-amber-500 hover:text-amber-300 disabled:opacity-50"

            >

              <Plus size={20} strokeWidth={3} /> ADD DRILL

            </button>

          </div>


          <div className="space-y-3">

            {timers.map((timer, index) => {

              const isCurrent = currentIndex === index;


              return (

                <div 

                  key={timer.id}

                  className={`group relative bg-neutral-900 border-l-8 rounded-r-lg p-5 transition-all ${

                    isCurrent ? 'border-amber-500 bg-neutral-800 shadow-lg' : 'border-neutral-700'

                  }`}

                >

                  <div className="relative flex items-center gap-4">

                    <div className={`w-10 h-10 flex items-center justify-center font-black italic text-xl ${

                      isCurrent ? 'text-amber-500' : 'text-neutral-600'

                    }`}>

                      #{index + 1}

                    </div>


                    <div className="flex-1 grid grid-cols-1 md:grid-cols-2 gap-2">

                      <input 

                        type="text" 

                        value={timer.label}

                        onChange={(e) => updateTimer(timer.id, 'label', e.target.value)}

                        disabled={isActive}

                        className="bg-transparent font-black text-lg focus:outline-none focus:text-amber-400 uppercase tracking-wider"

                        placeholder="DRILL NAME"

                      />

                      <div className="flex items-center gap-2">

                         <input 

                          type="number" 

                          value={timer.duration}

                          onChange={(e) => updateTimer(timer.id, 'duration', e.target.value)}

                          disabled={isActive}

                          className="w-20 bg-black text-amber-500 text-right font-black rounded border border-neutral-800 px-2 py-1 focus:outline-none focus:border-amber-500"

                        />

                        <span className="text-[10px] text-neutral-600 font-black">SECONDS</span>

                      </div>

                    </div>


                    <div className="flex items-center gap-2 md:opacity-0 group-hover:opacity-100 transition-opacity">

                      <div className="flex flex-col gap-1">

                        <button 

                          onClick={() => moveTimer(index, -1)}

                          disabled={isActive || index === 0}

                          className="p-1 text-neutral-600 hover:text-white disabled:opacity-0"

                        >

                          <ChevronUp size={20} />

                        </button>

                        <button 

                          onClick={() => moveTimer(index, 1)}

                          disabled={isActive || index === timers.length - 1}

                          className="p-1 text-neutral-600 hover:text-white disabled:opacity-0"

                        >

                          <ChevronDown size={20} />

                        </button>

                      </div>

                      <button 

                        onClick={() => duplicateTimer(timer.id)}

                        disabled={isActive}

                        title="Duplicate"

                        className="p-2 text-neutral-500 hover:text-amber-500 transition-colors"

                      >

                        <Copy size={18} />

                      </button>

                      <button 

                        onClick={() => removeTimer(timer.id)}

                        disabled={isActive || timers.length <= 1}

                        className="p-2 text-neutral-500 hover:text-red-500 transition-colors"

                      >

                        <Trash2 size={20} />

                      </button>

                    </div>

                  </div>

                </div>

              );

            })}

          </div>

        </div>


        {/* Preset Management */}

        <div className="mt-10 bg-neutral-900 border border-neutral-800 rounded-lg p-5">

          <h2 className="font-black text-neutral-400 text-lg tracking-widest mb-4 border-b border-neutral-800 pb-2 flex items-center gap-2">

            <Save size={20} /> SAVED WORKOUTS

          </h2>

          

          <div className="flex flex-col md:flex-row gap-4">

            <div className="flex-1 flex gap-2">

              <input 

                type="text"

                value={saveName}

                onChange={(e) => setSaveName(e.target.value)}

                placeholder="NAME THIS ROUTINE..."

                className="flex-1 bg-black border border-neutral-700 rounded px-4 py-3 text-sm font-bold focus:border-amber-500 outline-none text-white uppercase"

              />

              <button 

                onClick={saveSequence}

                disabled={!saveName.trim() || timers.length === 0}

                className="bg-neutral-800 hover:bg-neutral-700 text-amber-500 px-6 py-3 rounded font-black disabled:opacity-50 transition-colors flex items-center gap-2"

              >

                SAVE

              </button>

            </div>

            

            <div className="flex-1 relative">

              <select 

                onChange={(e) => {

                  if (e.target.value === 'default') {

                    loadSequence(DEFAULT_SEQUENCE);

                  } else if (e.target.value !== '') {

                    const seq = savedSequences.find(s => s.id === e.target.value);

                    if (seq) loadSequence(seq.timers);

                  }

                  e.target.value = ''; // Reset select after picking

                }}

                disabled={isActive}

                className="w-full bg-black border border-neutral-700 rounded px-4 py-3 text-sm font-bold text-amber-500 outline-none appearance-none cursor-pointer disabled:opacity-50 uppercase"

              >

                <option value="">LOAD A ROUTINE...</option>

                <option value="default">DEFAULT: 3 SETS</option>

                {savedSequences.map(seq => (

                  <option key={seq.id} value={seq.id}>{seq.name}</option>

                ))}

              </select>

              <div className="absolute right-4 top-1/2 -translate-y-1/2 pointer-events-none text-amber-500">

                <Download size={18} />

              </div>

            </div>

          </div>

        </div>


      </div>

    </div>

  );

};


export default App;
